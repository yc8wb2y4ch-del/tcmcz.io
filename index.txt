    let typeInterval;

    function render(id) {
        const node = script.find(s => s.id === id);
        if (!node) return;

        // 动态设置总结语
        if (id === 99) {
            if (patience >= 70) {
                node.text = "恭喜你！通过不懈的坚持，你最终成就了‘诗仙’的伟业。记住：只要功夫深，铁杵磨成针！";
            } else if (patience >= 30) {
                node.text = "看来，做到持之以恒并不容易。希望这次体验能让你明白，耐心和坚持是最宝贵的品质。";
            } else {
                node.text = "一时的顿悟如果没有长久的坚持，最终也可能归于平淡。希望这个故事能给你带来深刻的警示。";
            }
        }

        renderBase(node, () => showControls(node));
    }

    function renderBase(node, onComplete) {
        els.statusBar.style.display = node.hideStatus ? 'none' : 'flex';
        els.endingLayer.style.display = 'none';
        els.endingLayer.style.opacity = '0';
        els.choices.innerHTML = '';
        els.next.style.display = 'none';

        if (node.bg) els.scene.style.backgroundImage = `url('${node.bg}')`;

        if (node.showTitle) {
            els.title.innerText = node.showTitle;
            els.title.classList.remove('anim-title-show');
            void els.title.offsetWidth;
            els.title.classList.add('anim-title-show');
        }

        if (node.bubble) {
            els.bubble.innerText = node.bubble;
            els.bubble.style.display = 'block';
            els.bubble.className = 'bubble anim-bubble-show';
        } else {
            els.bubble.style.display = 'none';
        }

        // 重置层
        els.multiLayer.style.display = 'none';
        els.char.style.display = 'none';
        els.char.className = 'sprite';
        els.extra1.className = 'sub-sprite';

        // 特殊场景渲染
        if (node.char === "DUAL_SCENE_OLD_WOMAN" || node.char === "DUAL_SCENE_OLD_WOMAN_GRIND") {
            els.multiLayer.style.display = 'block';
            els.extra1.src = assets.char.liBaiSurprised; // 李白总是好奇/惊讶的
            els.extra1.style.left = '25%';
            els.extra1.style.transform = 'translateX(-50%)';
            els.extra1.style.display = 'block';

            els.char.src = (node.char === "DUAL_SCENE_OLD_WOMAN_GRIND") ? assets.char.oldWomanGrind : assets.char.oldWoman;
            els.char.style.left = '70%';
            els.char.style.transform = 'translateX(-50%) scaleX(-1)';
            els.char.style.display = 'block';
            if (node.char === "DUAL_SCENE_OLD_WOMAN_GRIND") {
                els.char.classList.add("anim-grind");
            } else {
                els.char.classList.add("anim-idle");
            }
        } else if (node.ending) {
            els.char.src = node.char;
            els.char.style.display = node.char ? 'block' : 'none';
            if (node.char) els.char.classList.add(node.anim || 'anim-idle');
            
            setTimeout(() => {
                els.endingLayer.style.display = 'flex';
                els.endingText.innerText = node.ending;
                setTimeout(() => els.endingLayer.style.opacity = '1', 50);
            }, 1000);

        } else if (node.char) {
            els.char.src = node.char;
            els.char.style.display = 'block';
            if (node.anim) els.char.classList.add(node.anim);
        }

        if (node.head) {
            els.portrait.src = node.head;
            els.portraitFrame.style.visibility = 'visible';
        } else {
            els.portraitFrame.style.visibility = 'hidden';
        }

        els.name.innerText = node.name || "系统";
        els.text.innerText = '';
        let i = 0;
        clearInterval(typeInterval);
        
        const completeTyping = () => {
            clearInterval(typeInterval);
            els.text.innerText = node.text;
            onComplete();
            i = node.text.length + 1;
        };
        
        els.box.onclick = null; // 重置点击事件
        if (!node.autoNext) {
            els.box.onclick = () => {
                if (i <= node.text.length) {
                    completeTyping();
                } else if (node.next) {
                    render(node.next);
                } else if (node.dynamicNext) {
                    render(getEnding());
                }
            };
        } else {
            // 自动过渡场景
            setTimeout(() => {
                if (node.dynamicNext) {
                    render(getEnding());
                } else {
                    render(node.next);
                }
            }, node.autoNext);
        }

        typeInterval = setInterval(() => {
            if (i >= node.text.length) {
                completeTyping();
            } else {
                els.text.innerText += node.text.charAt(i);
                i++;
            }
        }, 30);
    }

    function showControls(node) {
        if (node.choices) {
            node.choices.forEach(c => {
                let btn = document.createElement('button');
                btn.className = 'pixel-btn';
                btn.innerText = c.txt;
                btn.onclick = (e) => {
                    e.stopPropagation();
                    if (c.action) c.action();
                    if(c.tip) {
                        els.name.innerText = "系统提示";
                        els.text.innerText = c.tip;
                        els.choices.innerHTML = '';
                        els.next.style.display = 'block';
                        els.box.onclick = () => render(c.to);
                    } else {
                        render(c.to);
                    }
                };
                els.choices.appendChild(btn);
            });
        } else if (!node.autoNext && (node.next || node.dynamicNext || node.ending)) {
            els.next.style.display = 'block';
        }
    }

    window.restartGame = function() {
        patience = 50;
        updatePatience(0);
        render(1);
    }
    
    window.onload = () => render(1);
